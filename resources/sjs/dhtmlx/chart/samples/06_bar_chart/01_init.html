        view: "bar",
        container: "chart1",

        label: "#year#",
        width: 30,

        gradient: "3d",
        //border:false,
        width: 25,
        padding: {
            bottom: 0,
            right: 50,
            left: 50;
        }
    });
    barChart2.parse(data, "json");
}
