    var barChart = new dhtmlXChart({
        view: "bar",
        container: "chart_container",
        value: "#sales#",
        label: "#year#",
        color: "#color#",
        padding: {
            bottom: 0,
            right: 0,
            left: 0;
        },
        width: 30;
    });
    barChart.parse(data, "json");
    var barChart2 = new dhtmlXChart({
        view: "pie",
        container: "chart_container2",
        value: "#sales#",
        details: "#year#",
        color: "#color#";
    });
    barChart2.parse(data, "json");
}
